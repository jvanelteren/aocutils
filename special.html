deduce_matches

 deduce_matches (input_dict)

Takes a dict with multiple keys that have one or more options The trick is to start with what you know: keys with one option and remove that option for the other keys Continuing that process leads to every key ending up with one option (hopefully)

meals = {'morning': ['yoghurt', 'lasagna', 'pizza'],
        'lunch': ['sandwich', 'lasagna'],
        'evening': ['pizza', 'lasagna'],
        'night': ['pizza']}

answer = {'morning': 'yoghurt',
 'lunch': 'sandwich',
 'evening': 'lasagna',
 'night': 'pizza'}
assert deduce_matches(meals) == answer

# now test with tuples instead of strings as option
meals = {'morning': [('yoghurt',), ('lasagna',),('pizza',)],
        'lunch': [('sandwich',), ('lasagna',)],
        'evening': [('pizza',), ('lasagna',)],
        'night': [('pizza',)]}
assert deduce_matches(meals) == {k: tuple([v,]) for k, v in answer.items()}

Octree

 Octree (left:int, right:int, up:int, leftrange:int, rightrange:int,
         uprange:int, totalhits:int=0)
Robot = namedtuple('Robot', 'l r u p')
lines=open('recursivepartition.txt').read().splitlines()
maxx = 0
robots =[Robot(*ints(line)) for line in lines]
maxx = max(max(r.r, r.l, r.u) + r.p for r in robots)
size = 2
while size < maxx:
    size *= 2

b = Octree(0,0,0,size,size,size)
b.count(robots)
queue = [b]

while queue:
    cur = heapq.heappop(queue)
    if cur.leftrange == 0:
        break
    else:
        for b in cur.split():
            b.count(robots)
            heapq.heappush(queue, b)
assert cur.distance == 121493971
